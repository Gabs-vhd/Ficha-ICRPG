\newcommand{\smallfield}[1]{%
    \TextField[
        name=#1,
        width=1.2cm,
        height=0.4cm,
        bordercolor=black,
        backgroundcolor=white,
        charsize=14pt,
    ]{}%
}

    \TextField[
        name=#1,
        width=1.2cm,

}

\newcommand{\editbox}[2][4.5cm]{%
    \TextField[

        bordercolor=black,
        backgroundcolor=white,
        charsize=9pt,
        multiline=true
    ]{}%
}
